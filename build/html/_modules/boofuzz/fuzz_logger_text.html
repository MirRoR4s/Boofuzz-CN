    """

        :type bytes_to_str: function
        :param bytes_to_str: 用于将发送/接收的字节数据转换为字符串以进行日志记录，默认值为 DEFAULT_HEX_TO_STR
        """
        self._file_handle = file_handle
        self._format_raw_bytes = bytes_to_str
